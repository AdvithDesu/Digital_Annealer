\begin{frame}{GPU Thread Calculation in DA1}

    Each thread $i$ calculates the local field $L_i = \sum_j J_{ij}s_j + h_i$ and the energy difference is then calculated as $-2 \times s_i \times L_i$ depending on which $i^{\text{th}}$ spin flipped.

\begin{frame}{Pseudo-code for Digital Annealer 1 (DA1)}

    \begin{algorithm}[H]
\caption{Pseudo-code for Digital Annealer 1}
\begin{algorithmic}[1]
\Require Initial temperature, number of iterations $iters$, connectivity matrix $J[N][N]$ and magnetic field $h[N]$
\Ensure Spin configuration $s[N]$

\State \textbf{Initialization:} Randomly initialize spins $s[N]$, $k = 0$
\State Copy $J[N][N]$ and $h[N]$ to GPU memory

\While{$k \leq iters$}
    \For{$i = 1 : N$} \Comment{$N$ blocks for $N$-spin model}
        \For{$j = 1 : N/P$}
            \State Calculate partial sum of $\Delta H(s_i)$ by computing $J[i][j] \cdot s[j]$
        \EndFor

        \For{$p = 1 : \log(P)$} \Comment{Using $P$ threads}
            \State Reduction of partial sum of $\Delta H(s_i)$
        \EndFor

        \State Calculate $P_B(\Delta H, T)$ for $\Delta H(s_i)$ to determine state of spin $s_i$
    \EndFor \Comment{For all $N$ spins in ensemble}

    \State Update all possible spins
\EndWhile
\State Decrease temperature $T$ linearly

\end{algorithmic}
\end{algorithm}

    
\end{frame}

\begin{frame}[allowframebreaks]{Pseudo-code for DA1 with fractional flips}


\begin{algorithmic}[1]

\Require Initial temperature, number of iterations $iters$, connectivity matrix $J[N][N]$, magnetic field $h[N]$, flip fraction $\rho$
\Ensure Spin configuration $s[N]$

\State \textbf{Initialization:} Randomly initialize spins $s[N]$
\State Copy $J$, $h$, and $s$ to GPU memory

\While{$k \leq iters$}

    \State $A \gets \emptyset$ \Comment{List of Metropolis-accepted spins}

    % ----------- Phase 1: Compute candidates -----------
    \For{$i = 1 : N$ \textbf{in parallel}} \Comment{$N$ GPU blocks}

        \For{$j = 1 : N/P$} \Comment{$P$ threads per block}
            \State Compute partial sum of local field contribution:
            \[
                S_i \mathrel{+}= J[i][j] \cdot s[j]
            \]
        \EndFor

        \For{$p = 1 : \log(P)$} \Comment{Parallel reduction}
            \State Reduce partial sums to obtain full local field $S_i$
        \EndFor
        \State Compute energy change:
        \[
            \Delta H_i = -2\, s[i]\,(S_i + h[i])
        \]

        \State Compute Metropolis probability:
        \[
            P_B = \min\{1,\exp(-\Delta H_i/T)\}
        \]

        \If{random() $< P_B$}
            \State Append index $i$ to $A$
        \EndIf

    \EndFor
    \Comment{None of the spins are flipped}

    % ----------- Phase 2: Exact-k subset selection -----------
    \State Let $M \gets |A|$ \Comment{Total number accepted}

    \State $k \gets \lceil \rho \cdot M \rceil$ \Comment{Exact number of flips to perform}

    \State Randomly shuffle the list $A$ (Fisher--Yates)
    \State $A_k \gets$ first $k$ elements of $A$ \Comment{Random subset of size $k$}

    % ----------- Phase 3: Apply selected flips -----------
    \For{each $i \in A_k$ \textbf{in parallel}}
        \State Flip spin: $s[i] \gets -s[i]$
    \EndFor

    \State Update iteration counter: $k \gets k + 1$

\EndWhile

\State Decrease temperature $T$ linearly
\State \Return $s[N]$

\end{algorithmic}

\end{frame}

\section{Introduction}
%====================================================

\begin{frame}{Factorization Problem}
\begin{itemize}
\item Given a natural number $N$, find $P,Q$ such that
\[
N = P Q
\]
\item Hard inverse problem underlying RSA security.
\item Idea: map factorization $\rightarrow$ optimization problem.
\item Solve using annealing-based techniques.
\end{itemize}
\end{frame}

%====================================================
\section{Formulation}
%====================================================

\begin{frame}{Binary Representation and Assumptions}
Assumptions:
\begin{itemize}
\item $N$ is odd.
\item Factors have equal bit-length
\[
n=\left\lceil \tfrac12 \log_2 N \right\rceil.
\]
\end{itemize}

Binary form:
\[
P=\sum_{k=0}^{n-1}2^k p_k,\quad
Q=\sum_{k=0}^{n-1}2^k q_k,\quad p_k,q_k\in\{0,1\}.
\]

Fixed bits:
\[
p_0=q_0=1,\qquad p_{n-1}=q_{n-1}=1.
\]
\end{frame}

\begin{frame}{Column-wise Multiplication}
\begin{itemize}
\item Construct binary long multiplication table.
\item Introduce carry variables $s_{i,j}$.
\item Each column gives a constraint:
\[
C_i = N_i - \sum_{j} q_j p_{i-j}
      - \sum s_{j,i}
      + \sum 2^j s_{i,i+j}.
\]
\item Correct factors satisfy $C_i = 0$.
\end{itemize}
\end{frame}

\begin{frame}{Energy Function}
Optimization objective:
\[
\min_{\{p_i,q_i,s_{ij}\}} \sum_i (C_i)^2.
\]

\begin{itemize}
\item Binary variables: factor bits + carry bits.
\item Ground state corresponds to valid factorization.
\item Alternative direct method:
\[
\left(N-\sum p_i2^i \sum q_j2^j\right)^2
\]
but gives fewer preprocessing opportunities.
\end{itemize}
\end{frame}

%====================================================
\section{Classical Pre-processing}
%====================================================

\begin{frame}{Core Reduction Rules}
For binary variables $x_i\in\{0,1\}$:

\textbf{Sum rules}
\[
\sum_i x_i=n \Rightarrow x_i=1,\qquad
\sum_i x_i=0 \Rightarrow x_i=0.
\]

\textbf{Complement rule}
\[
x_1+x_2=1 \Rightarrow x_2=1-x_1.
\]

\textbf{Power rule}
\[
x^m=x,\quad m\ge2.
\]
\end{frame}

\begin{frame}{Coefficient and Parity Rules}
General linear constraint:
\[
\sum_i c_i x_i + C = 0.
\]

Let
\[
S_+=\sum_{c_i>0}c_i,\quad S_-=\sum_{c_j<0}|c_j|.
\]

\begin{itemize}
\item Large coefficients can force $x_i=0$ or $1$.
\item Parity constraints:
\begin{align*}
x_1+x_2+\text{even} &= \text{even} \Rightarrow x_1=x_2,\\
x_1+x_2+\text{even} &= \text{odd} \Rightarrow x_1+x_2=1.
\end{align*}
\end{itemize}
\end{frame}

\begin{frame}{Example: $N=15$ Reduction}
Initial clauses (example):
\begin{align*}
C_1 &: -p_1-q_1+2s_{1,2}+1=0,\\
C_2 &: -p_1q_1-p_2-q_2-s_{1,2}+2s_{2,3}+4s_{2,4}+1=0.
\end{align*}

Applying rules:
\begin{itemize}
\item Fix carries $s_{1,2}=s_{2,4}=0$.
\item Complement relation: $q_1=1-p_1$.
\item Power rule: $p_1^2=p_1$.
\end{itemize}

Final reduced constraint:
\[
2p_1p_2-p_1-p_2+1=0
\Rightarrow p_1+p_2=1.
\]
\end{frame}

\begin{frame}{Sparse Coupling Matrix Representation (CSR)}

\textbf{Problem:}
The Ising coupling matrix $J_{ij}$ is very large but mostly zero.
Storing it as a dense $N \times N$ matrix wastes memory and bandwidth.

\vspace{0.3cm}
\textbf{Key idea:}
Store \emph{only the nonzero couplings} using
\alert{Compressed Sparse Row (CSR)} format. CSR represents a sparse matrix using three arrays:

\[
\texttt{row\_ptr}[i] \quad \text{start index of row } i
\]
\[
\texttt{col\_idx}[k] \quad \text{column index of nonzero entry}
\]
\[
\texttt{values}[k] \quad \text{value of } J_{ij}
\]

Only nonzero entries are stored.

\vspace{0.2cm}
\textbf{Memory cost:} $\mathcal{O}(N + \text{nnz})$ instead of $\mathcal{O}(N^2)$

\end{frame}
